export class ParsedContentComponent {
  readonly content = input.required<string>();
  readonly citations = input<Citation[]>([]);

  private readonly docViewer = inject(DocumentViewerService);
  private readonly elRef = inject(ElementRef);
  private readonly destroyRef = inject(DestroyRef);

  // Tooltip state
  protected readonly tooltipVisible = signal(false);
  protected readonly tooltipCitation = signal<Citation | null>(null);
  protected readonly tooltipPos = signal({ top: 0, left: 0 });
  protected readonly tooltipAbove = signal(false);

  private hoverTimer: ReturnType<typeof setTimeout> | null = null;
  private leaveTimer: ReturnType<typeof setTimeout> | null = null;

  constructor() {
    this.destroyRef.onDestroy(() => {
      if (this.hoverTimer) clearTimeout(this.hoverTimer);
      if (this.leaveTimer) clearTimeout(this.leaveTimer);
    });
  }

  // Pre-computed tooltip display data — no Math/method calls in template
  protected readonly tooltipData = computed(() => {
    const c = this.tooltipCitation();
    if (!c) return null;
    return {
      sourceNumber: c.sourceNumber,
      title: c.title,
      snippet: c.snippet ? this.truncateSnippet(c.snippet) : null,
      domain: c.domain ?? null,
      domainLabel: c.domain ? getDomainLabel(c.domain) : null,
      domainIcon: c.domain ? getDomainIcon(c.domain) : null,
      tierLabel: getTierLabel(c.tier),
      confidence: c.confidence ? `${Math.round(c.confidence * 100)}%` : null,
      // Year-based YELLOW flag (Signal & Citator pattern)
      yearFlag: c.yearFlag ?? false,
      year: c.year ?? null,
    };
  });

  /**
   * Item 4: Merge all responseSegments from all citations into a flat list of
   * { start, end, nums[] } spans. Multiple citations that share the same byte span are merged.
   */
  protected readonly groundedSpans = computed(() => {
    type Span = { start: number; end: number; nums: number[] };
    const cites = this.citations();
    const spans: Span[] = [];

    for (const cite of cites) {
      if (!cite.responseSegments?.length) continue;
      for (const seg of cite.responseSegments) {
        const existing = spans.find(s => s.start === seg.startIndex && s.end === seg.endIndex);
        if (existing) {
          if (!existing.nums.includes(cite.sourceNumber ?? 0)) {
            existing.nums.push(cite.sourceNumber ?? 0);
          }
        } else {
          spans.push({ start: seg.startIndex, end: seg.endIndex, nums: [cite.sourceNumber ?? 0] });
        }
      }
    }
    return spans;
  });

  /**
   * Single computed: normalize → wrap grounded segments → inject citation badges.
   * Item 4: wrapGroundedSegments is called before injectCitationBadges so badge injection
   * doesn't interfere with the byte-offset span calculation.
   */
  protected readonly processedContent = computed(() => {
    const raw = this.content();
    const cites = this.citations();
    const spans = this.groundedSpans();
    const normalized = this.normalizeMarkdown(raw);
    const withUnderlines = this.wrapGroundedSegments(normalized, spans);
    return this.injectCitationBadges(withUnderlines, cites);
  });

  /**
   * Keyboard delegation: Enter/Space on .cite-badge triggers the same action as click.
   */
  onContainerKeyDown(event: KeyboardEvent) {
    if (event.key === 'Enter' || event.key === ' ') {
      const badge = (event.target as HTMLElement).closest?.('.cite-badge') as HTMLElement | null;
      if (badge) {
        event.preventDefault();
        this.onContainerClick(event as unknown as MouseEvent);
      }
    }
  }

  /**
   * Event delegation: catch clicks on .cite-badge elements within the rendered markdown.
   * Resolves the citation and opens the document viewer.
   */
  onContainerClick(event: MouseEvent) {
    const target = event.target as HTMLElement;
    // Use closest() to walk up DOM tree — handles cases where the click target
    // is a text node inside <sup>, or when markdown wraps the badge in other elements
    const badge = target.closest?.('.cite-badge') as HTMLElement | null;
    if (badge) {
      event.preventDefault();
      event.stopPropagation();
      const sourceNum = parseInt(badge.getAttribute('data-source-num') || '0', 10);
      const citation = this.resolveCitation(sourceNum);
      if (sourceNum > 0) {
        if (citation && citation.uri) {
          this.hideTooltip();
          // Item 1: prefer highlightText (full verbatim) over snippet (truncated)
          this.docViewer.openDocument(
            citation.uri,
            citation.title,
            citation.highlightText || citation.snippet || undefined
          );
        } else {
          console.warn('[ParsedContent] No citation found for source', sourceNum);
        }
      }
    }
  }

  /**
   * Hover delegation: show tooltip after 300ms debounce
   */
  onMouseOver(event: MouseEvent) {
    const target = event.target as HTMLElement;
    const badge = target.closest?.('.cite-badge') as HTMLElement | null;
    if (badge) {
      if (this.leaveTimer) {
        clearTimeout(this.leaveTimer);
        this.leaveTimer = null;
      }
      const sourceNum = parseInt(badge.getAttribute('data-source-num') || '0', 10);
      if (sourceNum > 0) {
        this.hoverTimer = setTimeout(() => {
          const citation = this.resolveCitation(sourceNum);
          if (citation) {
            this.showTooltip(badge, citation);
          }
        }, 300);
      }
    }
  }

  onMouseOut(event: MouseEvent) {
    const target = event.target as HTMLElement;
    const badge = target.closest?.('.cite-badge') as HTMLElement | null;
    if (badge) {
      if (this.hoverTimer) {
        clearTimeout(this.hoverTimer);
        this.hoverTimer = null;
      }
      // Small delay before hiding (prevents flicker when moving between badges)
      this.leaveTimer = setTimeout(() => {
        this.hideTooltip();
      }, 100);
    }
  }


  /**
   * Position and show the tooltip relative to the badge element.
   */
  private showTooltip(anchor: HTMLElement, citation: Citation) {
    const rect = anchor.getBoundingClientRect();
    const tooltipWidth = 380;
    const tooltipEstimatedHeight = citation.snippet ? 200 : 120;

    // Horizontal: center on anchor, clamp to viewport
    let left = rect.left + rect.width / 2 - tooltipWidth / 2;
    const padding = 16;
    if (left < padding) left = padding;
    // SSR guard: window is not available on server
    const viewportWidth = typeof window !== 'undefined' ? window.innerWidth : 1280;
    if (left + tooltipWidth > viewportWidth - padding) {
      left = viewportWidth - tooltipWidth - padding;
    }

    // Vertical: above by default, below if not enough space
    let top: number;
    let above = true;
    if (rect.top > tooltipEstimatedHeight + 20) {
      top = rect.top - tooltipEstimatedHeight - 8;
    } else {
      top = rect.bottom + 8;
      above = false;
    }

    this.tooltipCitation.set(citation);
    this.tooltipPos.set({ top, left });
    this.tooltipAbove.set(above);
    this.tooltipVisible.set(true);
  }

  private hideTooltip() {
    this.tooltipVisible.set(false);
    this.tooltipCitation.set(null);
  }

  // Template helpers
  truncateSnippet(snippet: string): string {
    return snippet.length > 250 ? snippet.substring(0, 250) + '...' : snippet;
  }

  getDomainLabel(domain: string): string {
    return DOMAIN_LABELS[domain] || domain;
  }

  getDomainIcon(domain: string): string {
    return DOMAIN_ICONS[domain] || 'description';
  }

  getTierLabel(tier?: number): string {
    return tier ? (TIER_LABELS[tier] || 'Reference') : 'Reference';
  }

  /**
   * Normalize LLM markdown output (matches Next.js normalizeMarkdown).
   */
  private normalizeMarkdown(content: string): string {
    let result = content;
    // Fix inline bullet points
    result = result.replace(/([.)\]\w])\s+\*\s+\*\*/g, '$1\n* **');
    // Fix inline numbered lists
    result = result.replace(/([.)\]])\s+(\d+)\.\s+\*\*/g, '$1\n$2. **');
    // Fix missing newlines before headings
    result = result.replace(/([^\n])(#{1,6}\s)/g, '$1\n$2');
    // Fix: LLM outputs bold label + table header on same line, e.g.:
    //   "**AUTHORITY HIERARCHY** | Source | Role |"
    //   "| :------ | :--- |"
    // marked can't parse this as a table because the first row lacks a leading "|"
    // and the bold text is inline with the header cells.
    // FIX: insert a blank line between the bold label and the pipe characters.
    result = result.replace(/(\*\*[^*]+\*\*)\s*(\|[^\n]+\|)/g, '$1\n\n$2');
    // Fix: Philippine legal LLM responses include footnote markers like [[FN 30]: text]]
    // These double-bracket patterns confuse the citation badge injector (Pass 2 matches
    // the inner [30] and injects a <sup> badge, but the outer [ stays, causing the
    // markdown parser to emit the badge's HTML attributes as raw visible text).
    // Strip them entirely — they are footnote labels already rendered inline in the text.
    result = result.replace(/\[\[FN\s+\d+\][^\]]*\]\]/g, '');
    result = result.replace(/\[\[FN\s+\d+\]\]/g, '');
    return result;
  }

  /**
   * Replace citation markers with clickable inline HTML <sup> badges.
   * Handles ALL formats the LLM may produce:
   *   [Source N], [Source N, p.5], [source_N], [N], ^[N],
   *   and Unicode superscript digits ¹²³⁴⁵⁶⁷⁸⁹
   *
   * Each badge gets data-source-num for event delegation AND data-tier for CSS coloring.
   */
  private injectCitationBadges(content: string, citations: Citation[]): string {
    const makeBadge = (num: number): string => {
      const c = citations.find(c => c.sourceNumber === num);
      const tier = c?.tier ? String(c.tier) : '';
      const tierAttr = tier ? ` data-tier="${tier}"` : '';
      // Inline styles guarantee rendering inside <markdown> where ::ng-deep may not reach
      // Sanitize: strip HTML tags, truncate, and escape double-quotes so the
      // aria-label attribute doesn't break out of its context (e.g. when citation
      // titles come from traverse_citation_graph tool results with embedded HTML).
      const rawTitle = c?.title ?? '';
      const safeTitle = rawTitle
        .replace(/<[^>]*>/g, '')           // strip HTML tags
        .replace(/&[a-z]+;/gi, ' ')        // collapse HTML entities
        .substring(0, 100)                 // max 100 chars
        .trim()
        .replace(/"/g, '&quot;');          // escape remaining quotes
      const title = safeTitle ? `: ${safeTitle}` : '';
      const style = [
        'display:inline-flex',
        'align-items:center',
        'justify-content:center',
        'min-width:20px',
        'height:20px',
        'padding:0 5px',
        'margin:0 2px',
        'background:rgba(0,0,0,0.08)',
        'color:#4b5563',
        'border:none',
        'border-radius:9999px',
        'font-size:0.7em',
        'font-weight:600',
        'cursor:pointer',
        'vertical-align:middle',
        'line-height:1',
        'text-decoration:none',
        'user-select:none',
      ].join(';');
      return `<sup class="cite-badge" role="button" tabindex="0" data-source-num="${num}"${tierAttr} aria-label="Source ${num}${title}" title="Source ${num}" style="${style}">${num}</sup>`;
    };

    // Spacer between adjacent badges — uses inline width for reliability
    const BADGE_SEP = '<span style="display:inline-block;width:2px"></span>';

    // Unicode superscript digit map
    const SUPER_MAP: Record<string, number> = {
      '\u00B9': 1, '\u00B2': 2, '\u00B3': 3,
      '\u2074': 4, '\u2075': 5, '\u2076': 6,
      '\u2077': 7, '\u2078': 8, '\u2079': 9,
    };

    let result = content;

    // Pass 0: Unicode superscript sequences (¹²³ etc.)
    // E.g. "damages ¹" or "evidence ²³"
    // Match one or more Unicode superscript digits, possibly space-separated
    result = result.replace(
      /[\u00B9\u00B2\u00B3\u2074-\u2079]+/g,
      (match) => {
        // Convert each superscript char to a badge
        return [...match].map(ch => {
          const num = SUPER_MAP[ch];
          return num ? makeBadge(num) : ch;
        }).join(BADGE_SEP);
      }
    );

    // Pass 1: [Source N, M, ...] or [source_N] — may contain multiple numbers
    // Handles ALL LLM variants:
    //   [source 1, 3]           → badge(1) badge(3)
    //   [source 1, source 3]   → badge(1) badge(3)
    //   [sources 1, 3, 5]      → badge(1) badge(3) badge(5)
    //   [source 4, source 5]   → badge(4) badge(5)
    result = result.replace(
      /\[sources?[\s_][\d,\s\w]+\]/gi,
      (match) => {
        const nums = [...match.matchAll(/\d+/g)].map(m => parseInt(m[0]));
        return nums.map(n => makeBadge(n)).join(BADGE_SEP);
      }
    );

    // Pass 2: bare [N] — only 1-99, not markdown link syntax [text](url)
    // Negative lookbehind (?<!\[) prevents matching [N] inside [[FN N]] double-bracket
    // patterns (the normalizeMarkdown step strips most, but belt-and-suspenders here too).
    result = result.replace(
      /(?<!\w)(?<!\[)\[(\d{1,2})](?!\()/g,
      (_, num) => makeBadge(parseInt(num))
    );

    // Pass 3: footnote-style ^[N]
    result = result.replace(
      /\^\[(\d{1,2})]/g,
      (_, num) => makeBadge(parseInt(num))
    );

    return result;
  }

  /**
   * Item 4: Wrap response text spans with .grounded-sentence elements.
   *
   * Byte offsets (UTF-8) from groundingSupports[].segment are converted to character
   * offsets using TextEncoder / TextDecoder so non-ASCII (Filipino diacritics) are safe.
   *
   * Processing order: descending by startIndex so splicing from the end doesn't
   * shift earlier character positions.
   *
   * Fallback: if spans is empty, returns text unchanged.
   */
  private wrapGroundedSegments(
    text: string,
    spans: Array<{ start: number; end: number; nums: number[] }>
  ): string {
    if (!spans.length) return text;

    // Encode full text to bytes once for efficient slicing
    const encoder = new TextEncoder();
    const decoder = new TextDecoder();
    const encoded = encoder.encode(text);

    // Sort descending so we splice from the end and don't shift prior offsets
    const sorted = [...spans].sort((a, b) => b.start - a.start);

    let result = text;
    for (const span of sorted) {
      // Convert byte slices back to string segments
      const before = decoder.decode(encoded.slice(0, span.start));
      const middle = decoder.decode(encoded.slice(span.start, span.end));
      const after = decoder.decode(encoded.slice(span.end));
      const numsAttr = span.nums.join(',');
      result = before
        + `<span class="grounded-sentence" data-source-nums="${numsAttr}" title="Grounded in source ${span.nums.join(', ')}">${middle}</span>`
        + after;
    }
    return result;
  }

  /**
   * Resolve a source number to a Citation object.
   */
  private resolveCitation(num: number): Citation | null {
    const found = this.citations().find(c => c.sourceNumber === num);
    return found || null;
  }
}
